CREATE TABLE CONTATO_MSG (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
EMAIL VARCHAR2(255) NOT NULL,
MENSAGEM VARCHAR(255) NOT NULL
);

CREATE TABLE ESTADO (
ID INTEGER PRIMARY KEY NOT NULL,
NOME CHAR(2) NOT NULL,
STATUS_DLT BOOLEAN
);

CREATE TABLE CIDADE (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ESTADO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ESTADO_FK) REFERENCES ESTADO(ID)
);

CREATE TABLE ENDERENCO (
ID INTEGER PRIMARY KEY NOT NULL,
RUA VARCHAR2(255) NOT NULL,
CEP CHAR(8) NOT NULL,
NUMERO VARCHAR2(10),
BAIRRO VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN,
ID_CIDADE_FK INTEGER NOT NULL,
FOREIGN KEY (ID_CIDADE_FK) REFERENCES CIDADE (ID)
);

CREATE TABLE CATEGORIA (
ID_CATEGORIA INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN
);

CREATE TABLE STATUS (
ID INTEGER PRIMARY KEY NOT NULL
DESCRICAO VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN
);

CREATE TABLE EMPRESA (
ID INTEGER PRIMARY KEY NOT NULL,
RAZAO_SOCIAL VARCHAR2(255) NOT NULL,
CNPJ CHAR(14) NOT NULL,
NOME_FANTASIA VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ENDERECO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ENDERECO_FK) REFERENCES ENDERENCO (ID)
);

CREATE TABLE CONTA_BANCAIA (
ID INTEGER PRIMARY KEY NOT NULL,
BANCO VARCHAR2(255),
AGENCIA VARCHAR2(6),
CONTA VARCHAR2(20),
STATUS_DLT BOOLEAN,
ID_EMP_FK INTEGER NOT NULL,
FOREIGN KEY (ID_EMP_FK) REFERENCES EMPRESA (ID)
);

CREATE TABLE RESPONSAVEL (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
CPF CHAR(11) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ENDERECO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ENDERECO_FK) REFERENCES ENDERENCO (ID)
);

CREATE TABLE BANNER (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
DATA_CADASTRO DATE NOT NULL,
DATA_EXPIRACAO DATE NOT NULL,
STATUS_DLT BOOLEAN,
ID_RESP_FK INTEGER NOT NULL,
FOREIGN KEY(ID_RESP_FK) REFERENCES RESPONSAVEL (ID)
);

CREATE TABLE FORNECEDOR (
ID INTEGER PRIMARY KEY NOT NULL,
RAZAO_SOCIAL VARCHAR2(255) NOT NULL,
NOME_FANTASIA VARCHAR2(255) NOT NULL,
CNPJ CHAR(11) NOT NULL,
STATUS_DLT BOOLEAN,
ID_END_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ENDERECO_FK) REFERENCES ENDERENCO (ID)
);

CREATE TABLE TELEFONE (
ID INTEGER PRIMARY KEY NOT NULL,
NUMERO VARCHAR2(20) NOT NULL,
ID_RESP_FK INTEGER,
FOREIGN KEY (ID_RESP_FK)
ID_EMP_FK INTEGER,
FOREIGN KEY(ID_EMP_FK) REFERENCES EMPRESA (ID),
ID_FORN_FK INTEGER,
FOREIGN KEY (ID_FORN_FK) REFERENCES FORNECEDOR (ID),
STATUS_DLT BOOLEAN,
)

CREATE TABLE BOLETO (
ID INTEGER PRIMARY KEY NOT NULL,
DESCRICAO VARCHAR2(255) NOT NULL,
DT_VENCIMENTO DATETIME NOT NULL,
VALOR NUMERIC(10,2) NOT NULL,
NUMERO_DOC VARCHAR2(255) NOT NULL,
COD_BARRA VARCHAR2(255),
ID_EMP_FK INTEGER NOT NULL,
FOREIGN KEY (ID_EMP_FK) REFERENCES EMPRESA (ID),
ID_FORN_FK INTEGER NOT NULL,
FOREIGN KEY(ID_FORN_FK) REFERENCES FORNECEDOR (ID),
ID_CATE_FK INTEGER NOT NULL,
FOREIGN KEY(ID_CATE_FK) REFERENCES CATEGORIA (ID),
STATUS_DLT BOOLEAN
)

CREATE TABLE PAGAMENTO (
ID INTEGER PRIMARY KEY NOT NULL,
DT_PAGAMENTO DATETIME NOT NULL,
DT_ENTREGA DATETIME NOT NULL,
JUROS NUMERIC(10,2),
ID_BOLE_FK INTEGER NOT NULL,
FOREIGN KEY(ID_BOLE_FK) REFERENCES BOLETO (ID),
ID_STATUS_FK INTEGER NOT NULL,
FOREIGN KEY(ID_STATUS_FK) REFERENCES STATUS (ID)
ID_RESP_FK INTEGER NOT NULL,
FOREIGN KEY(ID_RESP_FK) REFERENCES RESPONSAVEL (ID)
STATUS_DLT BOOLEAN
)

CREATE TABLE HIST_PAGAMENTO (
ID INTEGER PRIMARY KEY NOT NULL,
DT_PAGAMENTO DATETIME ,
DT_ENTREGA DATETIME,
JUROS NUMERIC(10,2),
ID_BOLE_FK INTEGER,
FOREIGN KEY(ID_BOLE_FK) REFERENCES BOLETO (ID),
ID_STATUS_FK INTEGER,
FOREIGN KEY(ID_STATUS_FK) REFERENCES STATUS (ID)
ID_RESP_FK INTEGER,
FOREIGN KEY(ID_RESP_FK) REFERENCES RESPONSAVEL (ID)
STATUS_DLT BOOLEAN
DATA_HORA DATETIME,
)

CREATE TABLE HIST_BOLETO (
ID INTEGER PRIMARY KEY NOT NULL,
DESCRICAO VARCHAR2(255),
DT_VENCIMENTO DATETIME ,
VALOR NUMERIC(10,2) ,
NUMERO_DOC VARCHAR2(255),
COD_BARRA VARCHAR2(255),
ID_EMP_FK INTEGER,
FOREIGN KEY (ID_EMP_FK) REFERENCES EMPRESA (ID),
ID_FORN_FK INTEGER,
FOREIGN KEY(ID_FORN_FK) REFERENCES FORNECEDOR (ID),
ID_CATE_FK INTEGER,
FOREIGN KEY(ID_CATE_FK) REFERENCES CATEGORIA (ID),
ID_BOLE_FK INTEGER NOT NULL,
FOREIGN KEY (ID_BOLE_FK) REFERENCES BOLETO (ID),
STATUS_DLT BOOLEAN
DATA_HORA DATETIME,
)
