CREATE TABLE CONTATO_MSG (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
EMAIL VARCHAR2(255) NOT NULL,
MENSAGEM VARCHAR(255) NOT NULL
);

CREATE TABLE ESTADO (
ID INTEGER PRIMARY KEY NOT NULL,
NOME CHAR(2) NOT NULL,
STATUS_DLT BOOLEAN
);

CREATE TABLE CIDADE (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ESTADO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ESTADO_FK) REFERENCES ESTADO(ID)
);

CREATE TABLE ENDERENCO (
ID INTEGER PRIMARY KEY NOT NULL,
RUA VARCHAR2(255) NOT NULL,
CEP CHAR(8) NOT NULL,
NUMERO VARCHAR2(10),
BAIRRO VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN,
ID_CIDADE_FK INTEGER NOT NULL,
FOREIGN KEY (ID_CIDADE_FK) REFERENCES CIDADE (ID)
);

CREATE TABLE CATEGORIA (
ID_CATEGORIA INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN
);

CREATE TABLE STATUS (
ID INTEGER PRIMARY KEY NOT NULL
DESCRICAO VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN
);

CREATE TABLE EMPRESA (
ID INTEGER PRIMARY KEY NOT NULL,
RAZAO_SOCIAL VARCHAR2(255) NOT NULL,
CNPJ CHAR(14) NOT NULL,
NOME_FANTASIA VARCHAR2(255) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ENDERECO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ENDERECO_FK) REFERENCES ENDERENCO (ID)
);

CREATE TABLE CONTA_BANCAIA (
ID INTEGER PRIMARY KEY NOT NULL,
BANCO VARCHAR2(255),
AGENCIA VARCHAR2(6),
CONTA VARCHAR2(20),
STATUS_DLT BOOLEAN,
ID_EMPRESA_FK INTEGER NOT NULL,
FOREIGN KEY (ID_EMPRESA_FK) REFERENCES EMPRESA (ID)
);

CREATE TABLE RESPONSAVEL (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
CPF CHAR(11) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ENDERECO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ENDERECO_FK) REFERENCES ENDERENCO (ID)
);

CREATE TABLE BANNER (
ID INTEGER PRIMARY KEY NOT NULL,
NOME VARCHAR2(255) NOT NULL,
DATA_CADASTRO DATE NOT NULL,
DATA_EXPIRACAO DATE NOT NULL,
STATUS_DLT BOOLEAN,
ID_RESPONSAVEL_FK INTEGER NOT NULL,
FOREIGN KEY(ID_RESPONSAVEL_FK) REFERENCES RESPONSAVEL (ID)
);

CREATE TABLE FORNECEDOR (
ID INTEGER PRIMARY KEY NOT NULL,
RAZAO_SOCIAL VARCHAR2(255) NOT NULL,
NOME_FANTASIA VARCHAR2(255) NOT NULL,
CNPJ CHAR(11) NOT NULL,
STATUS_DLT BOOLEAN,
ID_ENDERECO_FK INTEGER NOT NULL,
FOREIGN KEY(ID_ENDERECO_FK) REFERENCES ENDERENCO (ID)
);

CREATE TABLE TELEFONE (
ID INTEGER PRIMARY KEY NOT NULL,
NUMERO VARCHAR2(20) NOT NULL,
ID_RESP_FK INTEGER,
FOREIGN KEY (ID_RESP_FK)
ID_EMP_FK INTEGER,
FOREIGN KEY(ID_EMP_FK) REFERENCES EMPRESA (ID),
ID_FORN_FK INTEGER,
FOREIGN KEY (ID_FORN_FK) REFERENCES FORNECEDOR (ID),
STATUS_DLT BOOLEAN,
)

CREATE TABLE BOLETO (
ID INTEGER PRIMARY KEY NOT NULL,
DESCRICAO VARCHAR2(255) NOT NULL,
DT_VENCIMENTO DATETIME NOT NULL,
VALOR NUMERIC(10,2) NOT NULL,
NUMERO_DOC VARCHAR2(255) NOT NULL,
COD_BARRA VARCHAR2(255),
ID_EMP_FK INTEGER NOT NULL,
FOREIGN KEY (ID_EMP_FK) REFERENCES EMPRESA (ID),
ID_FORN_FK INTEGER NOT NULL,
FOREIGN KEY(ID_FORN_FK) REFERENCES FORNECEDOR (ID),
ID_CATE_FK INTEGER NOT NULL,
FOREIGN KEY(ID_CATE_FK) REFERENCES CATEGORIA (ID),
STATUS_DLT BOOLEAN
)

CREATE TABLE PAGAMENTO (
ID INTEGER PRIMARY KEY NOT NULL,
DT_PAGAMENTO DATETIME NOT NULL,
DT_ENTREGA DATETIME NOT NULL,
JUROS NUMERIC(10,2),
ID_BOLE_FK INTEGER NOT NULL,
FOREIGN KEY(ID_BOLE_FK) REFERENCES BOLETO (ID),
ID_STATUS_FK INTEGER NOT NULL,
FOREIGN KEY(ID_STATUS_FK) REFERENCES STATUS (ID)
ID_RESP_FK INTEGER,
FOREIGN KEY(ID_RESP_FK) REFERENCES RESPONSAVEL (ID)
STATUS_DLT BOOLEAN
)




CREATE TABLE HIST_PAGAMENTO (
DT_ENTREGA_HIST_PAGAMENTO VARCHAR(10),

ID_RESPONSAVEL_HIST_PAGAMENTO INTEGER,
ID_STATUS_HIST_PAGAMENTO INTEGER,
ID_HIST_PAGAMENTO INTEGER PRIMARY KEY,
ID_BOLETO_HIST_PAGAMENTO INTEGER,
ID_STATUS_HIST_PAGAMENTO1 INTEGER,
DATA_HORA DATETIME
)

STATUS_DLT_HIST_BOLETO VARCHAR(10),
VALOR_HIST_BOLETO VARCHAR(10),

ID_FORNECEDOR_HIST_BOLETO INTEGER,
ID_HIST_BOLETO INTEGER PRIMARY KEY,
ID_EMPRESA_HIST_BOLETO INTEGER,
DATA_HORA DATETIME
)

CREATE TABLE HIST_EMPRESA (
ID_HIST_EMP INTEGER PRIMARY KEY,
ID_EMPRESA__HIST_EMP INTEGER,
RAZAP_SOCIAL_HIST_EMP VARCHAR(10),

ID_ENDERECO_HIST_EMP INTEGER,
STATUS_DLT_HIST_EMP VARCHAR(10),
NOME_FANTASIA_HIST_EMP VARCHAR(10),
DATA_HORA VARCHAR(10),
ID_CONTA_BANCARIA_HIST_EMPRESA INTEGER,
ID_TELEFONE_HIST_EMPRESA INTEGER
)

CREATE TABLE HIST_FORNECEDOR (
ID_HIST_FORNECEDOR INTEGER PRIMARY KEY,
RAZAO_SOCIAL_HIST_FORNECEDOR VARCHAR(10),
CNPJ_HIST_FORNECEDOR VARCHAR(10),
NOME_FANTASIA_HIST_FORNECEDOR VARCHAR(10),
STATUS_DLT_HIST_FORNECEDOR VARCHAR(10),
ID_ENDERECO_HIST_FORNECEDOR INTEGER,
ID_TELEFONE_HIST_FORNECEDOR INTEGER
)

CREATE TABLE HIST_BANNER (
ID_HIST_BANNER INTEGER PRIMARY KEY,
ID_BANNER_HIST_BANNER INTEGER,
NOME_HIST_BANNER VARCHAR(10),
DATA_HORA VARCHAR(10),

)
